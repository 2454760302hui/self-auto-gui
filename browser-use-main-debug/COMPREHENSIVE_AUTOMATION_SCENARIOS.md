## 💻 代码示例

### 基础操作示例

```javascript
// 点击
element.click();

// 双击
const event = new MouseEvent('dblclick', { bubbles: true });
element.dispatchEvent(event);

// 右键点击
const rightClickEvent = new MouseEvent('contextmenu', { bubbles: true, button: 2 });
element.dispatchEvent(rightClickEvent);

// 悬停
const hoverEvent = new MouseEvent('mouseover', { bubbles: true });
element.dispatchEvent(hoverEvent);

// 长按
element.dispatchEvent(new MouseEvent('mousedown', { bubbles: true }));
setTimeout(() => {
    element.dispatchEvent(new MouseEvent('mouseup', { bubbles: true }));
}, 1000);

// 拖拽
const dragStartEvent = new DragEvent('dragstart', { bubbles: true });
element.dispatchEvent(dragStartEvent);
```

### 键盘操作示例

```javascript
// 输入文本
const input = document.getElementById('textInput');
input.value = '输入的文本';
input.dispatchEvent(new Event('input', { bubbles: true }));

// 按下按键
const keyDownEvent = new KeyboardEvent('keydown', { key: 'Enter', bubbles: true });
element.dispatchEvent(keyDownEvent);

// 组合键 (Ctrl+C)
const keyEvent = new KeyboardEvent('keydown', {
    key: 'c',
    ctrlKey: true,
    bubbles: true
});
element.dispatchEvent(keyEvent);
```

### 文本操作示例

```javascript
// 获取文本
const text = element.textContent;

// 设置文本
element.textContent = '新文本';

// 清空文本
element.textContent = '';

// 选中文本
const range = document.createRange();
range.selectNodeContents(element);
const selection = window.getSelection();
selection.removeAllRanges();
selection.addRange(range);

// 复制文本
const text = element.textContent;
navigator.clipboard.writeText(text);
```

### 属性操作示例

```javascript
// 获取属性
const attr = element.getAttribute('data-value');

// 设置属性
element.setAttribute('data-value', 'new-value');

// 移除属性
element.removeAttribute('data-value');

// 获取样式
const style = window.getComputedStyle(element).color;

// 设置样式
element.style.color = 'red';

// 获取类名
const className = element.className;
```
